country_event = {
	id = 78002
	title = "ctr_78002_title" # The Colonial Question
	desc = "ctr_78002_desc" # somethign about what we should do with their colonies

	is_triggered_only = yes
	major = yes

	# "We will have them for ourselves"
	# - Annex all of the colonial provinces
	# - Fires event 78003, Our Demands Met
	option = {
		name = "ctr_78002_opt1"

		FROM = {
			any_owned = {
				limit = {
					is_colonial = yes
				}
				secede_province = THIS
			}
		}

		country_event = 78003
	}

	# "Let them be free"
	# - Lose 2 infamy
	# - Releases all colonial states similar to dismantling
	#   - Each will have 200 relations with you
	# - Fires event 78003, Our Demands Met
	option = {
		name = "ctr_78002_opt2"

		badboy = -2

		FROM = {
			any_country = {
				limit = {
					have_cores_in = FROM
				}

				country_event = 78004
			}
		}

		country_event = 78003
	}
}

country_event = {
	id = 78003
	title = "ctr_78003_title" # Our Demands Met
	desc = "ctr_78003_desc"

	is_triggered_only = yes
	
	major = yes

	# "Huzzah"
	# - Annex any country we have a truce with who submitted to annexation
	# - Gain 30 prestige
	option = {
		name = "ctr_78003_opt1"

		any_country = {
			limit = {
				truce_with = THIS
				exists = yes
				has_country_flag = submitted_to_annexation
			}

			annex_to = THIS

			clr_country_flag = submitted_to_annexation
			clr_country_flag = declared_annexation
		}

		clr_country_flag = annexing_country

		prestige = 30
	}
}

country_event = {
	id = 78004
	title = "ctr_78004_title"
	desc = "ctr_78004_desc"

	allow_multiple_instances = yes

	is_triggered_only = yes

	option = {
		name = "ctr_78004_opt1"

		any_country = {
			limit = {
				exists = yes
				has_country_flag = submitted_to_annexation
			}

			any_owned = {
				limit = {
					is_colonial = yes
					is_core = THIS
				}

				secede_province = THIS
			}
		}

		# give relations and influence to the one who released us
		FROM = {
			relations = {
				who = THIS
				value = 400
			}

			diplomatic_influence = {
				who = THIS
				value = 100
			}
		}
	}
}

country_event = {
	id = 78005
	title = "ctr_78005_title" # "Our Failed Attempts"
	desc = "ctr_78005_desc"

	trigger = {
		# we tried to annex a country
		has_country_flag = annexing_country
		NOT = { has_country_flag = failed_annexation }

		# There's no country that was successfully annexed by us
		NOT = {
			any_neighbor_country = {
				has_country_flag = declared_annexation
				has_country_flag = submitted_to_annexation
				truce_with = THIS
			}
		}
	}

	mean_time_to_happen = {
		days = 5
	}

	# "We have failed $COUNTRY$"
	option = {
		name = "ctr_78005_opt1"

		# Remove the flag and add a new one for failing
		set_country_flag = failed_annexation
		clr_country_flag = annexing_country

		# Remove the declared_annexation flag from any country which has a truce with
		# us and didn't submit to annexation
		any_country = {
			limit = {
				has_country_flag = declared_annexation
				NOT = { has_country_flag = submitted_to_annexation }
				truce_with = THIS
			}

			clr_country_flag = declared_annexation
		}

		# lose 40% of prestige
		prestige_factor = -0.4

		# Gain 7 militancy and counsciousness
		any_pop = {
			consciousness = 3
		}
	}
}

country_event = {
	id = 78006
	title = "ctr_78006_title" # "Protests in $CAPITAL$"
	desc = "ctr_78006_desc"

	trigger = {
		has_country_flag = failed_annexation
	}

	mean_time_to_happen = {
		days = 30
	}

	# Put down these revolts with force
	option = {
		name = "ctr_78006_opt1"

		# kill off half of the people who would be in the protests
		capital_scope = {
			any_pop = {
				limit = {
					militancy = 5
				}

				reduce_pop = -0.50
			}

			# 10% of the soldiers from this province die putting down the revolt
			any_pop = {
				limit = {
					type = soldiers
				}

				reduce_pop = -0.10
			}
		}

		# increase the consciousness and militancy across the country 
		any_pop = {
			consciousness = 3
			militancy = -0.2
		}
	}

	# Let them speak their minds
	option = {
		name = "ctr_78006_opt2"

		public_meetings = yes_meeting

		any_pop = {
			consciousness = 2
			militancy = -0.5
		}
	}
}
