# 'Targetted by CTR' event (74000)
country_event = {

	id = 74000
	title = "ctr_74000_title"
	desc = "ctr_74000_desc"
	picture = "peele"

	# This will be triggered whenever the decision is taken
	is_triggered_only = yes

	# don't allow multiple countries to be targetted on the same day
	# just so that too many aren't targetted
	allow_multiple_instances = no

	# if this event occurs, immediately decrease the number
	# of countries which need to be targetted
	# Also, inform all countries running the campaign who
	# got targetted
	immediate = {
		change_variable = {
			which = ctr_num_countries_to_target
			value = -1
		}
	}

	# 'Damn' option
	option = {
		name = "ctr_74000_opt1"

		# we get targetted by CTR for 1 year (regardless of whether the campaign will
		# continue or not)
		add_country_modifier = {
			name = targetted_by_ctr
			duration = 365
		}

		any_country = {
			limit = {
				has_country_modifier = correcting_the_record
			}
			country_event = 74002
		}
	}

}

# 'Participate in Partition?' event (74001)
country_event = {

	id = 74001
	title = ctr_74001_title
	desc = ctr_74001_desc
	major = yes

	# This will be triggerd when
	#    we're a gp
	#    we're at war with a country being partitioned
	#    we're allied with the partition leader, who is inviting us
	trigger = {

		is_greater_power = yes
		NOT = {
			have_country_flag = participating_in_partition
			have_country_flag = denied_partition_request
		}

		# Make sure we're at war with the country being partitioned
		any_country = {
			war_with = THIS
			has_country_flag = being_partitioned
		}

		# Make sure we're allied to the partition leader
		any_country = {
			alliance_with = THIS
			has_country_flag = leading_partition
			has_country_flag = inviting_partitioners
		}
	}

	# This should happen pretty quickly after the leader invites people
	mean_time_to_happen = {
		days = 1
	}

	# "$COUNTRY$ will participate!"
	#    +2 infamy
	#    +10% prestige
	#    +100 relation points
	# 25% chance AI will choose
	option = {
		name = ctr_74001_opt1
		infamy = 2
		prestige_factor = 0.1

		# Ask the leading partitioner if they would allow us to join
		any_country = {
			limit = {
				alliance_with = THIS
				has_country_flag = leading_partition
			}
			country_event = 74002
		}

		ai_chance = {
			factor = 25
		}

		relation = {
			who = FROM
			value = 100
		}
	}

	# "We will not participate"
	#    -5% prestige
	#    -50 relation points
	# 75% chance AI will choose
	option = {
		name = ctr_74001_opt2
		prestige_factor = -0.05

		ai_chance = {
			factor = 75
		}

		relation = {
			who = FROM
			value = -50
		}
	}

}

# '$FROMCOUNTRY$ would like to participate!' event (74002)

	option = {

		}
	}

}
