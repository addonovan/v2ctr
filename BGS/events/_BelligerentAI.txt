# acquire state on a random neighboring country
country_event = {
	id = 75000
	title = "bgs_75000_title"
	desc = "bgs_75000_desc"

	trigger = {
		war = no
		civilized = yes
		NOT = { constructing_cb_type = acquire_state }
	}

	# mean time to happen
	mean_time_to_happen = {
		years = 5

		# 200% as likely to get it if they have unclaimed cores
		modifier = {
			factor = 2

			has_unclaimed_cores = yes
		}

		# 50% more likely if poor relations with a neighbor country
		modifier = {
			factor = 0.5

			any_neighbor_country = {
				NOT = { 
					relation = {
						who = THIS
						value = -100
					}
				}
			}
		}

		# 100% more likely if terrible relations iwth a neighboring country
		modfiier = {
			factor = 1.0

			any_neighbor_country = {
				NOT = {
					relation = {
						who = THIS
						value = -200
					}
				}
			}
		}
	}

	option = {
		name = "bgs_75000_opt1"

		random_country = {
			limit = {
				civilized = yes
				number_of_states = 2
				neighbour = THIS
				NOT = {
					relation = {
							who = THIS
							value = 0
						}
					}
				}
				add_casus_belli = {
					target = THIS
					type = acquire_state
					duration = 12
				}
			}
		}
	}
}
