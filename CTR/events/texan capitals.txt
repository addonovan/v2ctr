# The New Capital
country_event = {
	id = 78500
	title = "ctr_78500_title"
	desc = "ctr_78500_desc"

	trigger = {
		tag = TEX
		exists = yes
		war = no
		capital = 132

		NOT = { has_country_flag = columbia_not_renamed }
		NOT = { has_country_flag = capital_final }
	}

	mean_time_to_happen = {
		days = 14
	}

	# The capital should be in Houston
	option = {
		name = "ctr_78500_opt1"

		132 = {
			change_province_name = "Houston"
		}

		prestige = 5
	}

	# Keep it in Columbia
	option = {
		name = "ctr_78500_opt2"
		prestige = 2
		set_country_flag = columbia_not_renamed
	}
}
