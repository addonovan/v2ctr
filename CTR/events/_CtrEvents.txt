country_event = {
	id = 79000
	title = "ctr_79000_title"
	desc = "ctr_79000_desc"
	picture = "CTR_event"
	news = no

	trigger = {
		has_country_modifier = correcting_the_record
	}

	mean_time_to_happen = {
		months = 6

		# Every 25% of administrative spending will make the event 35% more likely to happen 

		modifier = {
			factor = 0.35
			administration_spending = 0.25
			NOT = { administration_spending = 0.50 }
		}

		modifier = {
			factor = 0.70
			administration_spending = 0.50
			NOT = { administration_spending = 0.75 }
		}

		modifier = {
			factor = 1.05
			administration_spending = 0.75
			NOT = { administration_spending = 1 }
		}

		modifier = {
			factor = 1.40
			administration_spending = 1
		}

		# 20% more likely when not at war

		modifier = {
			factor = 0.20
			war = no
		}

		# 25% bonus for civilized countries

		modifier = {
			factor = 0.25
			civilized = no
		}

	}

	# Target them
	option = {
		name = "ctr_79000_opt1"

		badboy = -3

		random_country = {
			badboy = 2

			random_list = {

				50 = {
					relation = {
						who = THIS
						value = -50
					}
				}

				25 = {
					relation = {
						who = THIS
						value = -75
					}
					prestige = -15
				}

				25 = {
					relation = {
						who = THIS
						value = -400
					}

					country_event = 79001
				}
			}
		}
	}

	# Perhaps we should find another target
	option = {
		name = "ctr_79000_opt2"
	}

}

country_event = {
	id = 79001
	title = "ctr_79001_title"
	desc = "ctr_79001_desc"
	picture = "CTR_event"
	news = no

	is_triggered_only = yes

	# We will not stand for this!
	option = {
		name = "ctr_79001_opt1"

		# Gain a Cut Down To Size cb against gp's and sp's
		random_owned = {
			limit = {
				FROM = {
					OR = {
						is_greater_power = yes
						is_secondary_power = yes
					}
				}
			}

			THIS = {
				casus_belli = {
					target = FROM
					type = cut_down_to_size
					months = 3
				}
			}
		}

		# Gain a Humiliate CB against "civilized" countries
		random_owned = {
			limit = {
				FROM = {
					is_greater_power = no
					is_secondary_power = no
					civilized = yes
				}
			}

			THIS = {
				casus_belli = {
					target = FROM
					type = humiliate
					months = 3
				}
			}
		}

		# Gain a "demand concession" against an unciv with >=2 states
		random_owned = {
			limit = {
				FROM = {
					civilized = no
					number_of_states = 2
				}
			}

			THIS = {
				casus_belli = {
					target = FROM
					type = demand_concession_casus_belli
					months = 3
				}
			}
		}

		# Gain a "establish protectorate" against an unciv with 1 state
		random_owned = {
			limit = {
				FROM = {
					civilized = no
					NOT = { number_of_states = 2 }
				}
			}

			THIS = {
				casus_belli = {
					target = FROM
					type = establish_protectorate_casus_belli
					months = 3
				}
			}
		}
	}

	# I'm sure they did not intend for this
	option = {
		name = "ctr_79001_opt2"

		prestige = -25
	}
}

country_event = {
	id = 79002
	title = "ctr_79002_title"
	desc = "ctr_79002_desc"
	picture = "CTR_event"
	news = no

	trigger = {
		has_country_modifier = correcting_the_record
		NOT = { has_country_modifier = funds_diverted }
	}

	mean_time_to_happen = {
		months = 8

		# 30% more likely if >5 million
		modifier = {
			factor = 0.30
			money = 5000000
		}

		# 20% more likely if >1 million
		modifier = {
			factor = 0.20
			money = 1000000
			NOT = { money = 5000000 }
		}

		# 10% more likely if >0.5 million
		modifier = {
			factor = 0.10
			money = 500000
			NOT = { money = 1000000 }
		}
	}

	# Yes, give them everything they need
	option = {
		name = "ctr_79002_opt1"

		# lose 62 days of tax income (were it completely maxed out)
		add_tax_relative_income = -62

		any_pop = {
			limit = {
				type = bureaucrat
			}
			money = 5000
		}

		add_country_modifier = {
			name = funds_diverted
			duration = 180
		}
	}

	# Spare the funds we can
	option = {
		name = "ctr_79002_opt2"

		# Lose 20 days of tax income (were it completely maxed out)
		add_tax_relative_income = -20

		any_pop = {
			limit = {
				type = bureaucrat
			}
			money = 1670
		}

		# Go into a cooldown for a little while
		add_country_modifier = {
			name = ctr_delayed
			duration = -1
		}
		remove_country_modifier = correcting_the_record
	}

	# This operation has grown too expensive
	option = {
		name = "ctr_79002_opt3"

		add_country_modifier = {
			name = ctr_cooldown
			duration = 180
		}
		remove_country_modifier = correcting_the_record
	}
}
