# 'Targetted by CTR' event (74000)
country_event = {

	id = 74000
	title = "ctr_74000_title"
	desc = "ctr_74000_desc"
	picture = "CTR_event"

	# This will be triggered whenever the decision is taken
	is_triggered_only = yes

	# don't allow multiple countries to be targetted on the same day
	# just so that too many aren't targetted
	allow_multiple_instances = no

	# if this event occurs, immediately decrease the number
	# of countries which need to be targetted
	# Also, inform all countries running the campaign who
	# got targetted
	immediate = {
		change_variable = {
			which = ctr_num_countries_to_target
			value = -1
		}
	}

	# 'Damn' option
	option = {
		name = "ctr_74000_opt1"

		# we get targetted by CTR for 1 year (regardless of whether the campaign will
		# continue or not)
		add_country_modifier = {
			name = targetted_by_ctr
			duration = 365
		}

		any_country = {
			limit = {
				has_country_modifier = correcting_the_record
			}
			country_event = 74002
		}
	}

}
