country_event = {
	id = 78001
	title = "ctr_78001_title" # Our Capitulation
	desc = "ctr_78001_desc"

	is_triggered_only = yes
	major = yes

	# "So be it"
	# - If hasColonies
	#   - Fire off event 78002. The Colonial Question
	# - Else
	#   - Fire off event 78003, Our Demands Met
	option = {
		name = "ctr_78001_opt1"

		random_country = {
			limit = {
				truce_with = THIS
				has_country_flag = annexing_country
			}
			country_event = 78002
		}
	}
}

country_event = {
	id = 78002
	title = "ctr_78002_title" # The Colonial Question
	desc = "ctr_78002_desc" # somethign about what we should do with their colonies

	is_triggered_only = yes
	major = yes

	# "We will have them for ourselves"
	# - Annex all of the colonial provinces
	# - Fires event 78003, Our Demands Met
	option = {
		name = "ctr_78002_opt1"

		FROM = {
			any_owned = {
				limit = {
					is_colonial = yes
				}
				secede_province = THIS
			}
		}

		country_event = 78003
	}

	# "Let them be free"
	# - Lose 2 infamy
	# - Releases all colonial states similar to dismantling
	#   - Each will have 200 relations with you
	# - Fires event 78003, Our Demands Met
	option = {
		name = "ctr_78002_opt2"

		badboy = -2

		FROM = {
			any_country = {
				limit = {
					have_cores_in = FROM
				}

				country_event = 78004
			}
		}

		country_event = 78003
	}
}

country_event = {
	id = 78003
	title = "ctr_78003_title" # Our Demands Met
	desc = "ctr_78003_desc"

	is_triggered_only = yes
	
	major = yes

	# "Huzzah"
	# - Annex any country we have a truce with who submitted to annexation
	# - Gain 30 prestige
	option = {
		name = "ctr_78003_opt1"

		any_country = {
			limit = {
				truce_with = THIS
				exists = yes
				has_country_flag = submitted_to_annexation
			}

			annex_to = THIS

			clr_country_flag = submitted_to_annexation
			clr_country_flag = declared_annexation
		}

		clr_country_flag = annexing_country

		prestige = 30
	}
}

country_event = {
	id = 78004
	title = "ctr_78004_title"
	desc = "ctr_78004_desc"

	allow_multiple_instances = yes

	is_triggered_only = yes

	option = {
		name = "ctr_78004_opt1"

		any_country = {
			limit = {
				exists = yes
				has_country_flag = submitted_to_annexation
			}

			any_owned = {
				limit = {
					is_colonial = yes
					is_core = THIS
				}

				secede_province = THIS
			}
		}

		# give relations and influence to the one who released us
		FROM = {
			relations = {
				who = THIS
				value = 400
			}

			diplomatic_influence = {
				who = THIS
				value = 100
			}
		}
	}
}

country_event = {
	id = 78005
	title = "ctr_78005_title" # "Our Failed Attempts"
	desc = "ctr_78005_desc"

	trigger = {
		# we tried to annex a country
		has_country_flag = annexing_country

		# There's no country that was successfully annexed by us
		NOT = {
			any_country = {
				has_country_flag = declared_annexation
				has_country_flag = submitted_to_annexation
				truce_with = THIS
			}
		}
	}

	mean_time_to_happen = {
		days = 5
	}

	# "We have failed $COUNTRY$"
	option = {
		name = "ctr_78005_opt1"

		# Gain 7 militancy and counsciousness
		militancy = 7
		consciousness = 7

		# lose 40% of prestige
		prestige_factor = -0.4

		any_pop = {
			limit = {
				type = soldiers
			}

			scaled_militancy = 5
			scaled_consciousness = 5

			random_list = {
				75 = {
					ideology = {
						factor = 0.50
						value = liberal
					}
				}

				25 = {
					ideology = {
						factor = 0.50
						value = communist
					}
				}
			}
		}
	}
}
